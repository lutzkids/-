pragma solidity 0.8.0; // 指定Solidity编译器版本为0.8.0

contract TarkovBTC { // 定义合约名称为TarkovBTC
    string public name = "Tarkov BTC"; // 定义代币名称为Tarkov BTC
    string public symbol = "SMNJT"; // 定义代币符号为SMNJT
    uint8 public decimals = 8; // 定义小数位数为8位
    uint256 public totalSupply = 1000000000 * 10**8; // 定义总供应量为100亿，考虑8位小数
    
    uint256 public constant TAX_BASIS_POINTS = 5; // 定义税费比例为5个基点（即0.05%）
    uint256 public constant BASIS_POINTS_DENOMINATOR = 10000; // 定义基点分母为10000（代表100%）
    
    address public creator; // 定义合约创建者地址，用于接收税费，支持后续更换
    
    uint256 public maxTxAmount = totalSupply / 100; // 定义单笔最大交易限额，默认为总供应量的1%（即1亿枚）
    
    mapping(address => uint256) private _balances; // 定义地址到余额的映射表
    mapping(address => mapping(address => uint256)) private _allowances; // 定义地址到授权额度的嵌套映射表
    mapping(address => bool) public exemptFromTax; // 定义免税白名单映射，标记哪些地址可免除转账税费
    
    event Transfer(address indexed from, address indexed to, uint256 value); // 定义转账事件
    event Approval(address indexed owner, address indexed spender, uint256 value); // 定义授权事件
    event TaxRecipientChanged(address indexed previousRecipient, address indexed newRecipient); // 定义税费接收地址变更事件
    event ExemptStatusChanged(address indexed account, bool isExempt); // 定义免税白名单状态变更事件
    event MaxTxAmountChanged(uint256 previousLimit, uint256 newLimit); // 定义最大交易限额变更事件
    
    modifier onlyCreator() { // 定义权限修饰符，限制仅合约创建者可调用特定函数
        require(msg.sender == creator, "Caller is not the creator"); // 检查调用者是否为当前创建者地址
        _; // 如果权限验证通过，继续执行被修饰的函数体
    }
    
    constructor() { // 构造函数，在合约部署时执行一次
        creator = msg.sender; // 将合约部署者地址记录为初始创建者
        _balances[msg.sender] = totalSupply; // 将全部代币余额分配给合约部署者
        exemptFromTax[msg.sender] = true; // 将创建者地址加入免税白名单，避免向自己转账时重复扣税
        emit Transfer(address(0), msg.sender, totalSupply); // 触发转账事件，标记代币从空地址铸造
    }
    
    function balanceOf(address account) public view returns (uint256) { // 查询指定地址的代币余额
        return _balances[account]; // 返回该地址在映射表中记录的余额数值
    }
    
    function allowance(address owner, address spender) public view returns (uint256) { // 查询授权额度
        return _allowances[owner][spender]; // 返回所有者授权给支出者可以使用的代币数量
    }
    
    function setExemptFromTax(address account, bool isExempt) public onlyCreator { // 设置指定地址的免税白名单状态，仅创建者可调用
        exemptFromTax[account] = isExempt; // 更新映射表中该地址的免税状态
        emit ExemptStatusChanged(account, isExempt); // 触发事件，将状态变更记录上链
    }
    
    function setMaxTxAmount(uint256 newMaxTxAmount) public onlyCreator { // 修改单笔最大交易限额，仅创建者可调用
        require(newMaxTxAmount >= totalSupply / 1000, "MaxTxAmount too low"); // 检查新限额不得低于总供应量的0.1%，防止过度限制交易
        uint256 previousLimit = maxTxAmount; // 记录当前限额数值，用于事件通知
        maxTxAmount = newMaxTxAmount; // 更新最大交易限额为新的数值
        emit MaxTxAmountChanged(previousLimit, newMaxTxAmount); // 触发事件，记录限额变更历史
    }
    
    function transferTaxRecipient(address newRecipient) public onlyCreator { // 更换税费接收地址，仅创建者可调用
        require(newRecipient != address(0), "New recipient is zero address"); // 检查新地址是否为空地址，防止税费转入黑洞
        address previousRecipient = creator; // 记录原税费接收地址，用于事件通知
        creator = newRecipient; // 将创建者变量更新为新的税费接收地址
        emit TaxRecipientChanged(previousRecipient, newRecipient); // 触发事件，记录接收地址变更历史
    }
    
    function _transfer(address from, address to, uint256 amount) internal { // 内部转账函数，处理核心转账逻辑
        require(from != address(0), "Transfer from zero address"); // 检查发送地址是否为空地址，防止无效转账
        require(to != address(0), "Transfer to zero address"); // 检查接收地址是否为空地址，防止代币丢失
        require(_balances[from] >= amount, "Insufficient balance"); // 检查发送方余额是否充足，不足则回滚交易
        require(amount <= maxTxAmount, "Exceeds max transaction amount"); // 检查转账金额是否超过单笔最大限额，防止大额操纵
        
        uint256 finalAmount = amount; // 定义实际转账金额变量，初始化为原始转账金额
        uint256 taxAmount = 0; // 定义税费金额变量，初始化为零
        
        if (!exemptFromTax[from] && !exemptFromTax[to]) { // 判断发送方和接收方是否均不在免税白名单中
            taxAmount = (amount * TAX_BASIS_POINTS) / BASIS_POINTS_DENOMINATOR; // 计算转账税费金额，按0.05%比例收取
            finalAmount = amount - taxAmount; // 计算接收方实际收到的代币数量（扣除税费后）
        }
        
        _balances[from] -= amount; // 从发送方余额中扣除全部转账金额（包含税费）
        _balances[to] += finalAmount; // 向接收方余额增加实际到账金额（已扣税或全额）
        
        if (taxAmount > 0) { // 判断税费金额是否大于零，避免零值转账浪费gas
            _balances[creator] += taxAmount; // 将税费金额累加到合约创建者的余额中
            emit Transfer(from, creator, taxAmount); // 触发转账事件，记录税费流向创建者
        }
        
        emit Transfer(from, to, finalAmount); // 触发转账事件，记录实际转账金额流向接收方
    }
    
    function transfer(address to, uint256 amount) public returns (bool) { // 标准ERC20转账函数，直接调用者向他人转账
        _transfer(msg.sender, to, amount); // 调用内部转账函数，发送方为当前调用者
        return true; // 返回true表示转账成功执行
    }
    
    function transferFrom(address from, address to, uint256 amount) public returns (bool) { // 授权转账函数，由被授权方代为转账
        uint256 currentAllowance = _allowances[from][msg.sender]; // 查询当前调用者被授权可使用的额度
        require(currentAllowance >= amount, "Allowance exceeded"); // 检查授权额度是否足够，不足则回滚交易
        
        _allowances[from][msg.sender] = currentAllowance - amount; // 扣除已使用的授权额度
        _transfer(from, to, amount); // 调用内部转账函数执行实际转账操作
        return true; // 返回true表示授权转账成功执行
    }
    
    function approve(address spender, uint256 amount) public returns (bool) { // 授权函数，允许指定地址使用调用者的代币
        require(spender != address(0), "Approve to zero address"); // 检查被授权地址是否为空地址，防止授权失效
        
        _allowances[msg.sender][spender] = amount; // 在映射表中记录调用者授权给支出者的额度
        emit Approval(msg.sender, spender, amount); // 触发授权事件，记录授权信息上链
        return true; // 返回true表示授权成功执行
    }
    
    function increaseAllowance(address spender, uint256 addedValue) public returns (bool) { // 增加授权额度函数，在原有基础上追加
        uint256 newAllowance = _allowances[msg.sender][spender] + addedValue; // 计算新的授权额度（原额度加上追加额度）
        _allowances[msg.sender][spender] = newAllowance; // 更新映射表中的授权额度为新的数值
        emit Approval(msg.sender, spender, newAllowance); // 触发授权事件，记录更新后的授权额度
        return true; // 返回true表示增加授权额度成功
    }
    
    function decreaseAllowance(address spender, uint256 subtractedValue) public returns (bool) { // 减少授权额度函数，在原有基础上削减
        uint256 currentAllowance = _allowances[msg.sender][spender]; // 查询当前授权额度
        require(currentAllowance >= subtractedValue, "Decreased allowance below zero"); // 检查削减金额是否超过原额度，防止下溢
        
        _allowances[msg.sender][spender] = currentAllowance - subtractedValue; // 更新映射表中的授权额度为削减后的数值
        emit Approval(msg.sender, spender, _allowances[msg.sender][spender]); // 触发授权事件，记录更新后的授权额度
        return true; // 返回true表示减少授权额度成功
    }
}
